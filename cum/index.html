Momentum conservation: flow tends to continue in same direction let momentumScore = 0; if (i > 1) { const prevIndex = path[path.length - 2].price; const currentIndex = path[path.length - 1].price; const prevDirection = currentIndex > prevIndex ? 1 : (currentIndex < prevIndex ? -1 : 0); const nextDirection = j > currentPriceIndex ? 1 : (j < currentPriceIndex ? -1 : 0); // Higher score if maintaining direction momentumScore = prevDirection === nextDirection ? 3.0 : 0; } // 3. Path of least resistance: prefer flowing around walls than through them const resistanceScore = nextPressure.isWall ? -5.0 : 0; // 4. Flow velocity alignment: prefer paths aligned with the velocity field const flowDirection = Math.sign(nextVelocity.vy); const movementDirection = j > currentPriceIndex ? 1 : (j < currentPriceIndex ? -1 : 0); const alignmentScore = flowDirection === movementDirection ? 2.0 : -1.0; // 5. Actual price attraction: slight bias toward actual market price const actualPriceScore = actualPriceIndex >= 0 ? (1.0 - Math.abs(j - actualPriceIndex) / priceLevels.length) * 1.0 : 0; // Calculate total flow score for this potential next point const totalFlowScore = pressureScore + momentumScore + resistanceScore + alignmentScore + actualPriceScore; // Update best next point if this has a better flow score if (totalFlowScore > maxFlowScore) { maxFlowScore = totalFlowScore; nextPriceIndex = j; } } // Move to the next position with highest flow score currentPriceIndex = nextPriceIndex; } // Calculate fluid dynamics metrics for visualization and prediction function calculateFluidDynamics(flowPath, pressureField, velocityField) { const velocityVectors = []; const confidenceScores = [];
